Install-Package Serilog
Install-Package Serilog.Sinks.File
Install-Package Serilog.Extensions.Logging


using System;
using Serilog;

namespace ConsoleLoggingExample
{
    class Program
    {
        static void Main(string[] args)
        {
            // Configure Serilog
            Log.Logger = new LoggerConfiguration()
                .MinimumLevel.Information()
                .WriteTo.File("Logs/log.txt", 
                              rollingInterval: RollingInterval.Day,
                              outputTemplate: "[{Timestamp:yyyy-MM-dd HH:mm:ss}] {Message}{NewLine}")
                .CreateLogger();

            try
            {
                // Log messages
                Log.Information("Application started");
                PerformOperation();
                Log.Information("Application ended");
            }
            catch (Exception ex)
            {
                Log.Error(ex, "An error occurred while running the application");
            }
            finally
            {
                Log.CloseAndFlush();
            }
        }

        static void PerformOperation()
        {
            Log.Information("Performing operation...");
            // Simulate some work
            System.Threading.Thread.Sleep(1000); // Delay to mimic operation
            Log.Information("Operation completed successfully.");
        }
    }
}
