using System;
using System.Windows.Forms;

namespace Win32ListViewApp
{
    public partial class MainForm : Form
    {
        public MainForm()
        {
            InitializeComponent();
        }

        private void MainForm_Load(object sender, EventArgs e)
        {
            // Initialize ListView properties
            listViewItems.View = View.List;  // Set ListView to list view mode
            listViewItems.FullRowSelect = true;  // Ensure full row is selected
        }

        // Button click event to add items to the ListView
        private void btnAdd_Click(object sender, EventArgs e)
        {
            if (!string.IsNullOrWhiteSpace(txtNewItem.Text))
            {
                listViewItems.Items.Add(txtNewItem.Text);
                txtNewItem.Clear();  // Clear input field after adding item
            }
            else
            {
                MessageBox.Show("Please enter a valid item.", "Input Error", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            }
        }

        // Button click event to remove selected item from the ListView
        private void btnRemove_Click(object sender, EventArgs e)
        {
            if (listViewItems.SelectedItems.Count > 0)
            {
                listViewItems.Items.Remove(listViewItems.SelectedItems[0]);  // Remove selected item
            }
            else
            {
                MessageBox.Show("Please select an item to remove.", "Selection Error", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            }
        }
    }
}
