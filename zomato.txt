<?xml version="1.0" encoding="UTF-8"?>
<restaurants>
    <restaurant>
        <name>Pizza Place</name>
        <address>123 Main St</address>
        <rating>4.5</rating>
    </restaurant>
    <restaurant>
        <name>Burger Joint</name>
        <address>456 Elm St</address>
        <rating>4.0</rating>
    </restaurant>
</restaurants>


using System;
using System.Linq;
using System.Xml.Linq;

class ZomatoLikeApp
{
    static void Main(string[] args)
    {
        // Load the XML file
        string xmlFilePath = "restaurants.xml";  // Path to the XML file
        XDocument xmlDoc = XDocument.Load(xmlFilePath);

        // Parse and display restaurant data
        var restaurants = xmlDoc.Descendants("restaurant");

        foreach (var restaurant in restaurants)
        {
            string name = restaurant.Element("name")?.Value;
            string address = restaurant.Element("address")?.Value;
            string rating = restaurant.Element("rating")?.Value;

            Console.WriteLine($"Restaurant: {name}, Address: {address}, Rating: {rating}");
        }
    }
}
